mars_god = {
	scope = character 

	effect = {
		set_variable = {
			name = deity
			value = flag:mars
		}
		remove_character_modifier = venus_god
		remove_character_modifier = mercury_god
		remove_character_modifier = minerva_god
		remove_character_modifier = laverna_god
		add_character_modifier = mars_god
	}
}
