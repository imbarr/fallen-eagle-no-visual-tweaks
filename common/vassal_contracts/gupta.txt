	display_mode = checkbox
	obligation_levels = {

		}
		gupta_nobility_scutage_on = {
			parent = gupta_nobility_scutage_none
			position = { 1 0 }
			is_valid = {
				custom_description = {
					text = gupta_nobility_scutage_validity

						AND = {
							trigger_if = {
								limit = {
									exists = father
								}
								father = {
									is_lowborn = no
								}
							}
							trigger_else_if = {
								limit = {

							}

								always = no
							}

								limit = {
									exists = mother
								}

								}
							}
							trigger_else_if = {
								limit = {
									NOT = {
										exists = mother
									}
								}

							}
							trigger_else = {
								always = no
							}
						}
					}
				}
			}
			
			liege_opinion = -30
